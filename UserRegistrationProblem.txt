echo "Welcome to User Registration Problem"

function FirstName()
{
read -p "Enter first name: " fname
if [[ $fname =~ $NamePattern ]]
then
	echo "First Name is correct"
else
	echo "First name should start with Capital, try again"
	FirstName
fi
}

function LastName()
{
read -p "Enter Last name: " lname
if [[ $lname =~ $NamePattern ]]
then
   echo "Last Name is correct"
else
   echo "Last name should start with Capital, try again"
   LastName
fi
}

NamePattern="^[[:upper:]]{1}[[:lower:]]{3,}"
FirstName
LastName
